<body>

    <h1>Typing Challenge Game</h1>

    <p>Upload a word, then type it within 5 seconds to win!</p>


    <input type="text" id="word-input" placeholder="Enter a word" />

    <button id="upload-button">Upload Word</button>


    <div class="game-container">

        <div id="game" style="display: none;">

            <p class="word-display" id="word-display">Word will appear here</p>

            <input type="text" id="typing-input" class="input-field" placeholder="Type the word here" />

            <p id="timer">Time left: 5 seconds</p>

        </div>

        <p class="message" id="message"></p>

    </div>


    <script>

        const uploadButton = document.getElementById('upload-button');

        const wordInput = document.getElementById('word-input');

        const wordDisplay = document.getElementById('word-display');

        const game = document.getElementById('game');

        const typingInput = document.getElementById('typing-input');

        const timerDisplay = document.getElementById('timer');

        const message = document.getElementById('message');


        let word = "";

        let timeLeft = 5;

        let timer;


        uploadButton.addEventListener('click', () => {

            word = wordInput.value.trim();


            if (!word) {

                alert('Please enter a valid word.');

                return;

            }


            wordInput.value = "";

            wordDisplay.textContent = word;

            game.style.display = 'block';

            typingInput.value = "";

            typingInput.focus();

            message.textContent = "";


            startTimer();

        });


        typingInput.addEventListener('input', () => {

            if (typingInput.value.trim() === word) {

                clearInterval(timer);

                message.textContent = "You win! 🎉";

                resetGame();

            }

        });


        function startTimer() {

            timeLeft = 5;

            timerDisplay.textContent = `Time left: ${timeLeft} seconds`;


            timer = setInterval(() => {

                timeLeft--;

                timerDisplay.textContent = `Time left: ${timeLeft} seconds`;


                if (timeLeft <= 0) {

                    clearInterval(timer);

                    message.textContent = "Time's up! You lose! 💔";

                    resetGame();

                }

            }, 1000);

        }


        function resetGame() {

            game.style.display = 'none';

        }

    </script>
